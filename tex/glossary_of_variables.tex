\chapter*{Glossary of variables and functions}
\begin{acronym}[MinHash]
    \acro{n}{size of a sketch}
    \acro{t}{threshold value at which}
    \acro{b}{number of bucket functions $g_j(p)$ in non-amplified LSH scheme}
    \acro{r}{number of hash functions $h_i(p)$ contributing to single bucket functions $g_j(p)$ in non-amplified LSH scheme}
    \acro{z}{weighted average of $Pr[\text{False positive}]$ and $Pr[\text{False negative}]$}
    \acro{rho}[$\mathbf{\rho}$]{measure for effectiveness of LSH scheme}
    \acro{k}{number of bins in \textit{one permuation hashing} (OPH) scheme}
\end{acronym}
